--------------------------------
-- Criar o Esquema Relacional
--------------------------------
-- DROP VIEW IF EXISTS V_

DROP TABLE IF EXISTS gps_ponto;
DROP TABLE IF EXISTS terreno;
DROP TABLE IF EXISTS tipo_terreno;
---------------------------------



---------------------------------
-- TIPO_TERRENO
---------------------------------
CREATE TABLE tipo_terreno (
    id_tipo_terreno INT PRIMARY KEY,
    nome VARCHAR(30) UNIQUE NOT NULL
);



---------------------------------
-- TERRENO
---------------------------------
CREATE TABLE terreno (
    id_terreno INT PRIMARY KEY,
    id_tipo_terreno INT REFERENCES tipo_terreno(id_tipo_terreno),
    g_terreno GEOMETRY(POLYGON, 4326)
    
);



---------------------------------
-- GPS_PONTO
---------------------------------
CREATE TABLE gps_ponto (
id_ordem INT,
id_terreno INT,
ponto GEOMETRY(POINT, 4326)
);

ALTER TABLE gps_ponto
ADD CONSTRAINT pk_gps_ponto PRIMARY KEY (id_ordem, id_terreno);
ALTER TABLE gps_ponto
ADD CONSTRAINT Fk_gps_ponto FOREIGN KEY (id_terreno) REFERENCES terreno (id_terreno);
